│   ├── .env                     # Приватные ключи бэкенда (GIGACHAT_AUTH_KEY). Игнорируется Git.
│   │
│   ├── database/                # ЯДРО ДАННЫХ (Твоя зона ответственности)

│   │   ├── models.py            # Декларативные ORM-модели (структура таблиц)
│   │   └── crud.py              # Функции манипуляции данными (get, create, bulk_insert)
│   │
│   ├── api/                     # СЛОЙ СЕТЕВЫХ ЭНДПОИНТОВ (Маршруты для Axios)

│   │   ├── schemas.py           # Pydantic-схемы (жесткие контракты для входящего JSON)
│   │   └── endpoints/           # HTTP-контроллеры
│   │       ├── __init__.py
│   │       ├── transactions.py  # Эндпоинты истории и ручного ввода (/api/transactions)
│   │       └── ai.py            # Эндпоинты импорта: /api/import/text, /api/import/image
│   │
│   └── services/                # СЛОЙ ИНТЕГРАЦИЙ (AI-модуль)
│       ├── __init__.py
│       ├── text_AI.py           # Логика извлечения данных из текста через GigaChat
│       └── OCR_AI.py            # Логика распознавания изображений (Tesseract + GigaChat)
